JavaScript
document.querySelectorAll('.cm-frost-nav').forEach(function (nav) {
  var group = nav.querySelector('.cm-frost-nav__links');
  var links = Array.from(group.querySelectorAll('a'));
  var current = group.querySelector('[aria-current="page"]') || links[0];

  function movePill(link) {
    group.style.setProperty('--cm-pill-x', link.offsetLeft + 'px');
    group.style.setProperty('--cm-pill-width', link.offsetWidth + 'px');
  }

  links.forEach(function (link) {
    link.addEventListener('pointerenter', function () { movePill(link); });
    link.addEventListener('focus', function () { movePill(link); });
  });
  group.addEventListener('pointerleave', function () { movePill(current); });
  nav.addEventListener('focusout', function (event) {
    var next = event.relatedTarget;
    if (!(next instanceof Node) || !nav.contains(next)) movePill(current);
  });
  if ('ResizeObserver' in window) {
    new ResizeObserver(function () { movePill(current); }).observe(group);
  } else {
    window.addEventListener('resize', function () { movePill(current); });
  }
  requestAnimationFrame(function () { movePill(current); });
});

JavaScript
var reducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)');

document.querySelectorAll('.cm-glass-card-wrap').forEach(function (wrap) {
  var frame;

  function reset() {
    cancelAnimationFrame(frame);
    wrap.classList.remove('is-active');
    wrap.style.setProperty('--cm-tilt-x', '0deg');
    wrap.style.setProperty('--cm-tilt-y', '0deg');
  }

  wrap.addEventListener('pointermove', function (event) {
    if (reducedMotion.matches || event.pointerType === 'touch') return;
    cancelAnimationFrame(frame);
    frame = requestAnimationFrame(function () {
      var rect = wrap.getBoundingClientRect();
      if (!rect.width || !rect.height) return;
      var x = Math.max(0, Math.min(1, (event.clientX - rect.left) / rect.width));
      var y = Math.max(0, Math.min(1, (event.clientY - rect.top) / rect.height));
      wrap.classList.add('is-active');
      wrap.style.setProperty('--cm-tilt-x', ((0.5 - y) * 12).toFixed(2) + 'deg');
      wrap.style.setProperty('--cm-tilt-y', ((x - 0.5) * 12).toFixed(2) + 'deg');
      wrap.style.setProperty('--cm-glare-x', (x * 100).toFixed(1) + '%');
      wrap.style.setProperty('--cm-glare-y', (y * 100).toFixed(1) + '%');
    });
  });
  wrap.addEventListener('pointerleave', reset);
  wrap.addEventListener('pointercancel', reset);
  wrap.addEventListener('focusout', function (event) {
    var next = event.relatedTarget;
    if (!(next instanceof Node) || !wrap.contains(next)) reset();
  });
});
